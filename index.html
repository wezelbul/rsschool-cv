public class DigPow {
    public static long digPow(int n, int p) {
        int sum = 0;
        int pow = p;
        for (char ch : String.valueOf(n).toCharArray()) {
            sum += Math.pow(Character.getNumericValue(ch), pow);
            pow += 1;
        }
        if (sum % n == 0) {
            return (long) sum / n;
        } else {
            return -1;
        }
    }
}
